Creating a local server using Express
==============================

import express from 'express'
const app = express();
const port = process.env.PORT || 4000

// Get Request to home page
app.get("/", (req, res)=>{
	res.send("This is home page")
})

// listening on port
app.listen(port, ()=>{
	log("Server start at http://localhost:"+port)
})
